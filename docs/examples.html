02_create_actor.msq

[init]
actor a Alice

[scenario]
a -* b label=create actor_label=Bob
a -> b stop
b + end

03_box.msq

[init]
actor a Alice
actor ""
actor b Bob

[scenario]
b -box ready
a -> b SYN
b -box "computing...\n(step 1)"
a -box waiting...
b -box "computing...\n(step 2)"
a <- b SYN,ACK
a -> b ACK
